// 1. 三項演算子の省略形（null合体演算子）
$name = $_GET['name'] ?? 'Guest';
// isset($_GET['name']) ? $_GET['name'] : 'Guest' の短縮版
// 存在チェックと代入をまとめられて便利


// 2. compact()で変数から連想配列を作る
$name = 'げん';
$age = 25;
$data = compact('name', 'age');
// ['name' => 'げん', 'age' => 26]
// 同名の変数をキーにした配列を一発で生成できる


// 3. ヒアドキュメントで長い文字列をそのまま書ける
$html = <<<HTML
<div class="box">
    <p>Hello, PHP!</p>
</div>
HTML;
// 複数行のHTMLやメール本文を組み立てたい時に便利


// 4. array_filter() で false な要素を除外
$filtered = array_filter(['a', '', null, 'b']);
// 結果: ['a', 'b']
// 空文字や null を一括で取り除ける。第二引数で独自条件も指定できる


// 5. DateTime で柔軟な日付処理
$now = new DateTime();
echo $now->modify('+3 days')->format('Y-m-d');
// 日付計算やフォーマットを安全に扱える
// strtotime()より高機能で、タイムゾーン指定も可能
