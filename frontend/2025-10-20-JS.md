# 偶奇判定
```
<p><input type="text" id="number" placeholder="数値を入力"></p>
<button onclick="clickNow3()">判定する</button>
<p id="answer"></p>

<script>
  function clickNow3() {
    const val = document.getElementById('number').value.trim();
    const answer = document.getElementById('answer');

    if (val === '') {
      answer.textContent = '数値を入れてね';
      return;
    }

    const n = Number(val);
    if (!Number.isFinite(n)) {
      answer.textContent = '正しい数値を入れてね';
      return;
    }

    answer.textContent = (n % 2 === 0) ? '偶数' : '奇数';
  }

  // ついでにEnterキーでも判定できるように
  document.getElementById('number').addEventListener('keydown', (e) => {
    if (e.key === 'Enter') clickNow3();
  });
</script>
```
